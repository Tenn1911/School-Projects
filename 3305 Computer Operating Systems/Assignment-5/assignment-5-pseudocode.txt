            if(current_capacity == 0){
                break;

            }
            
        }
        else{
            current_capacity = drop_off_passengers(current_station, current_capacity, train_num)
            if(current_capacity == 0){

                current_station -= 1

                current_station += 1
            }
        }
    }
    
    //end of the thread
    return NULL
}
